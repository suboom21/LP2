def guess_number():
    while True:
        try:
            guess = int(input("Guess the number (1-6): "))
            if 1 <= guess <= 6:
                return guess
            else:
                print("Please enter a number between 1 and 6.")
        except ValueError:
            print("Please enter a valid number.")
