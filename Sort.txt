x = []
n = int(input("How many elements do you want to sort? "))


for i in range(n):
    a = int(input("Enter"))
    x.append(a)
print("Unsorted elements: ", x)


for i in range(0, n-1):
    for j in range(i+1, n):
        if x[i] > x[j]:
            x[i], x[j] = x[j], x[i]
print("Sorted elements: ", x)
